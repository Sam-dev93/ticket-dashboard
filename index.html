function formatCurrency(amount) {


return new Intl.NumberFormat('en-GB', {


style: 'currency',


currency: 'GBP',


minimumFractionDigits: 0


}).format(Math.abs(amount));

function formatDate(dateStr) {


const [day, month, year] = dateStr.split('/');


const date = new Date(year, month - 1, day);


return date.toLocaleDateString('en-GB', {


day: 'numeric',


month: 'short'


});


}




function isWithinNextWeek(dateStr) {


const [day, month, year] = dateStr.split('/');


const matchDate = new Date(year, month - 1, day);


const today = new Date();


const nextWeek = new Date(today.getTime() + 7 * 24 * 60 * 60 * 1000);


return matchDate >= today && matchDate <= nextWeek;


}

async function getDocumentInfo(sheetId) {


const url = `https://sheets.googleapis.com/v4/spreadsheets/${sheetId}?key=${API_KEY}`;




try {


const response = await fetch(url);


if (!response.ok) {


throw new Error(`Failed to fetch sheet metadata: ${response.status}`);


}


const data = await response.json();


return {


title: data.properties.title,


sheets: data.sheets || []


};


} catch (error) {


console.error('Error fetching sheet metadata:', error);


throw error;


}


}




async function fetchSheetData(sheetId, range = 'A1:Z50') {


const url = `https://sheets.googleapis.com/v4/spreadsheets/${sheetId}/values/${range}?key=${API_KEY}`;




try {


const response = await fetch(url);


if (!response.ok) {


throw new Error(`Failed to fetch sheet data: ${response.status}`);


}


const data = await response.json();


return data.values || [];


} catch (error) {


console.error('Error fetching sheet data:', error);


throw error;


}


}




async function fetchMatchDetails(sheetId, matchName) {


try {


// Try to fetch the specific match sheet


const matchData = await fetchSheetData(sheetId, `${matchName}!A1:Z50`);


if (!matchData) return null;




// Extract date from cell R1 (row 1, column R = index 17)


let matchDate = matchData[0] && matchData[0][17] ? matchData[0][17] : null;




// Parse different date formats


if (matchDate) {


// Handle various date formats from Google Sheets


if (typeof matchDate === 'string') {


// Try to parse formats like "13/09/2025" or "13 Sep 2025"


const dateFormats = [


/(\d{1,2})\/(\d{1,2})\/(\d{4})/, // DD/MM/YYYY


/(\d{1,2})\s+(\w+)\s+(\d{4})/ // DD MMM YYYY


];




for (const format of dateFormats) {


const match = matchDate.match(format);


if (match) {


if (format === dateFormats[0]) {


// DD/MM/YYYY format


matchDate = `${match[1].padStart(2, '0')}/${match[2].padStart(2, '0')}/${match[3]}`;


} else {


// DD MMM YYYY format - convert to DD/MM/YYYY


const months = {


'jan': '01', 'feb': '02', 'mar': '03', 'apr': '04',


'may': '05', 'jun': '06', 'jul': '07', 'aug': '08',


'sep': '09', 'oct': '10', 'nov': '11', 'dec': '12'


};


const monthNum = months[match[2].toLowerCase().substr(0, 3)] || '01';


matchDate = `${match[1].padStart(2, '0')}/${monthNum}/${match[3]}`;


}


break;


}


}


}


}




// Count tickets from rows 4+ (index 3+)


let totalTickets = 0;


let soldTickets = 0;




for (let i = 3; i < matchData.length; i++) {


const row = matchData[i];


if (!row) continue;




const fvValue = row[2] ? row[2].toString().trim() : ''; // Column C (FV)


const soldFor = row[11] ? row[11].toString().trim() : ''; // Column L (SOLD FOR)




if (fvValue && fvValue.includes('£')) {


totalTickets++;


// Check if sold - SOLD FOR column has a value and it's not the header


if (soldFor && soldFor !== '' && !soldFor.toLowerCase().includes('sold for') && soldFor.includes('£')) {


soldTickets++;


}


}


}




return {


date: matchDate || '01/08/2025',


totalTickets: totalTickets,


soldTickets: soldTickets,


availableTickets: totalTickets - soldTickets


};




} catch (error) {


console.log(`Could not fetch details for ${matchName}:`, error.message);


return null;


}


}




function parseSheetData(rows, sheetName, documentTitle) {


if (!rows || rows.length === 0) return null;




let totalTickets = 0;


let totalSpent = 0;


let totalSales = 0;


let netProfit = 0;




// Look for dashboard data in the sheet


for (let i = 0; i < rows.length; i++) {


const row = rows[i];


if (!row || !row[1]) continue;




const cell = row[1].toString().trim();


const value = row[2] ? row[2].toString().trim() : '';




if (cell === 'TOTAL TICKETS') {


totalTickets = parseInt(value) || 0;


} else if (cell === 'TOTAL SPENT (FV)') {


totalSpent = parseFloat(value.replace(/[£,]/g, '')) || 0;


} else if (cell === 'TOTAL SALES') {


totalSales = parseFloat(value.replace(/[£,]/g, '')) || 0;


} else if (cell === 'NET PROFIT') {


netProfit = parseFloat(value.replace(/[£,]/g, '')) || 0;


}


}




// If no dashboard data found, this might be a match sheet, skip it


if (totalTickets === 0 && totalSpent === 0) {


return null;


}




const matches = [];


let matchTableStartRow = -1;




// Find match table


for (let i = 0; i < rows.length; i++) {


const row = rows[i];


if (!row) continue;




if (row[1] && row[1].toString().includes('MATCH') &&


row[2] && row[2].toString().includes('TICKETS') &&


row[3] && row[3].toString().includes('SPENT') &&


row[4] && row[4].toString().includes('SALES')) {


matchTableStartRow = i + 1;


break;


}


}




if (matchTableStartRow > 0) {


for (let i = matchTableStartRow; i < rows.length; i++) {


const row = rows[i];


if (!row || !row[1]) continue;




const matchName = row[1].toString().trim();


const tickets = parseInt(row[2]) || 0;


const spent = parseFloat(row[3]?.toString().replace(/[£,]/g, '')) || 0;


const sales = parseFloat(row[4]?.toString().replace(/[£,]/g, '')) || 0;


const profit = parseFloat(row[5]?.toString().replace(/[£,]/g, '')) || (sales - spent);




if (!matchName.includes('V') && !matchName.includes('v')) continue;


if (tickets === 0) continue;




// Calculate sold tickets based on sales vs spending


const soldTickets = sales > 0 ? Math.min(tickets, Math.round(sales / (spent / tickets))) : 0;


const availableTickets = tickets - soldTickets;




matches.push({


name: matchName,


date: '01/08/2025', // Will be updated from individual sheet


tickets: availableTickets,


total: tickets,


sold: soldTickets,


spent: spent,


sales: sales,


profit: profit


});


}


}




// Use document title as team name, fall back to sheet name


let teamName = documentTitle || sheetName;




// Remove "BOUGHT 25-26" and similar patterns from the title


teamName = teamName.replace(/\s+BOUGHT\s+\d{2}-\d{2}$/i, '').trim();




return {


name: teamName,


emoji: getTeamEmoji(teamName),


totalTickets: totalTickets,


totalSpent: totalSpent,


totalSales: totalSales,


netProfit: netProfit,


matches: matches


};


}

async function refreshData() {


const loadingElement = document.getElementById('loadingMessage');


const refreshBtn = document.querySelector('.refresh-btn');




loadingElement.style.display = 'block';


refreshBtn.disabled = true;


refreshBtn.textContent = '🔄 Refreshing...';




try {


const allTeams = [];




// Process each Google Sheets document in parallel for speed


const documentPromises = SHEET_IDS.map(async (sheetId) => {


console.log(`Processing document: ${sheetId}`);




try {


// Get document info and all sheets


const documentInfo = await getDocumentInfo(sheetId);


const documentTitle = documentInfo.title;


const sheets = documentInfo.sheets;




console.log(`Document: "${documentTitle}" has ${sheets.length} sheets`);




// Find the main dashboard sheet (usually the first one that's not "seasons")


const dashboardSheet = sheets.find(sheet =>


!sheet.properties.title.toLowerCase().includes('season')


);




if (!dashboardSheet) {


console.log(`No dashboard sheet found in ${documentTitle}`);


return null;


}




const sheetName = dashboardSheet.properties.title;


console.log(`Using dashboard sheet: ${sheetName} for ${documentTitle}`);




// Fetch data from this sheet


const sheetData = await fetchSheetData(sheetId, `${sheetName}!A1:F50`);




// Try to parse as team dashboard


const teamData = parseSheetData(sheetData, sheetName, documentTitle);




if (teamData) {


// Update match details with real data from individual sheets for accuracy


for (const match of teamData.matches) {


try {


const matchDetails = await fetchMatchDetails(sheetId, match.name);


if (matchDetails) {


match.date = matchDetails.date;


match.total = matchDetails.totalTickets;


match.sold = matchDetails.soldTickets;


match.tickets = matchDetails.availableTickets;


console.log(`Updated ${match.name}: ${match.sold}/${match.total} sold, date: ${match.date}`);


}


} catch (error) {


console.log(`Could not update match details for ${match.name}:`, error.message);


// Keep dashboard data as fallback


}


}




console.log(`Successfully parsed team: ${teamData.name}`);


return teamData;


}




return null;


} catch (error) {


console.error(`Error processing document ${sheetId}:`, error);


return null;


}


});




// Wait for all documents to be processed


const results = await Promise.all(documentPromises);




// Filter out null results


const validTeams = results.filter(team => team !== null);




if (validTeams.length === 0) {


throw new Error('No valid team data found in any sheets');


}




console.log(`Found ${validTeams.length} teams total`);




// Calculate totals across all teams


const totals = validTeams.reduce((acc, team) => {


acc.totalTickets += team.totalTickets;


acc.totalSpent += team.totalSpent;


acc.totalSales += team.totalSales;


acc.netProfit += team.netProfit;


// Calculate available tickets


const availableTickets = team.matches.reduce((sum, match) => sum + match.tickets, 0);


acc.availableTickets += availableTickets;


return acc;


}, { totalTickets: 0, availableTickets: 0, totalSpent: 0, totalSales: 0, netProfit: 0 });




// Update dashboard data


dashboardData = {


...totals,


teams: validTeams


};




updateDashboard();




refreshBtn.textContent = '✅ Updated!';


console.log('Dashboard updated successfully');




setTimeout(() => {


refreshBtn.textContent = '🔄 Refresh Data';


refreshBtn.disabled = false;


}, 2000);




} catch (error) {


console.error('Error refreshing data:', error);


refreshBtn.textContent = '❌ Error - Try Again';




let errorMsg = 'Failed to load data. ';


if (error.message.includes('403')) {


errorMsg += 'Make sure all sheets are set to "Anyone with link can view".';


} else if (error.message.includes('400')) {


errorMsg += 'Check that the sheet IDs are correct.';


} else {


errorMsg += error.message;


}




alert(errorMsg);




setTimeout(() => {


refreshBtn.textContent = '🔄 Refresh Data';


refreshBtn.disabled = false;


}, 3000);


} finally {


loadingElement.style.display = 'none';


}


}




updateDashboard();


setInterval(refreshData, 5 * 60 * 1000);


</script>
